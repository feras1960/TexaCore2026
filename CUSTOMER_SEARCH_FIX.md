### 6. الكود المحسّن

#### loadAllTenants()
```typescript
// 1. جلب المشتركين مع المنتجات (LEFT JOIN)
const { data: tenantsData } = await supabase
  .from('tenants')
  .select(`
    *,
    saas_products (id, name, code)
  `)
  .in('status', ['active', 'trial', 'pending', 'suspended'])
  .order('name');

// 2. جلب الاشتراك لكل عميل بشكل منفصل
for (const tenant of tenantsData) {
  const { data: subscription } = await supabase
    .from('tenant_subscriptions')
    .select(`
      id, status, start_date, end_date,
      subscription_plans (*)
    `)
    .eq('tenant_id', tenant.id)
    .in('status', ['active', 'trial'])
    .limit(1)
    .maybeSingle();
  
  // 3. دمج البيانات
  tenantsWithDetails.push({
    ...tenant,
    product_name: tenant.saas_products?.name,
    subscription: subscription
  });
}
```

```typescript
useEffect(() => {
  if (!searchQuery.trim()) {
    setFilteredTenants(tenants);
    return;
  }

  const q = searchQuery.toLowerCase();
  const filtered = tenants.filter(tenant =>
    tenant.name?.toLowerCase().includes(q) ||
    tenant.code?.toLowerCase().includes(q) ||
    tenant.email?.toLowerCase().includes(q) ||
    tenant.phone?.toLowerCase().includes(q) ||
    tenant.contact_person?.toLowerCase().includes(q) ||
    tenant.tax_number?.toLowerCase().includes(q) ||
    tenant.product_name?.toLowerCase().includes(q)
  );

  setFilteredTenants(filtered);
}, [searchQuery, tenants]);
```

