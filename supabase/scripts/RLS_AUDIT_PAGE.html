📊 استعلام الفحص السريع

-- ===================================================== -- AUDIT_quick.sql - فحص RLS السريع -- تاريخ: 2026-02-05 -- ===================================================== -- 1. الملخص العام SELECT '📊 ملخص النظام' as report, (SELECT COUNT(*) FROM pg_tables WHERE schemaname = 'public') as total_tables, (SELECT COUNT(*) FROM pg_policies WHERE schemaname = 'public') as total_policies, (SELECT COUNT(*) FROM information_schema.triggers WHERE trigger_schema = 'public') as total_triggers; -- 2. حالة RLS لكل جدول (أول 50 جدول) SELECT t.tablename as table_name, CASE WHEN c.relrowsecurity THEN '✓' ELSE '✗' END as rls_enabled, COALESCE(p.cnt, 0) as policies_count, CASE WHEN EXISTS (SELECT 1 FROM information_schema.columns WHERE table_schema='public' AND table_name=t.tablename AND column_name='tenant_id') THEN '✓' ELSE '-' END as has_tenant_id, CASE WHEN EXISTS (SELECT 1 FROM information_schema.columns WHERE table_schema='public' AND table_name=t.tablename AND column_name='company_id') THEN '✓' ELSE '-' END as has_company_id FROM pg_tables t LEFT JOIN pg_class c ON t.tablename = c.relname AND c.relnamespace = 'public'::regnamespace LEFT JOIN (SELECT tablename, COUNT(*) cnt FROM pg_policies WHERE schemaname='public' GROUP BY tablename) p ON t.tablename = p.tablename WHERE t.schemaname = 'public' ORDER BY t.tablename LIMIT 50;

📝 استعلام تفصيلي للمشاكل

-- استعلام كشف المشاكل -- ═══════════════════════════════════════════════════════════════ -- 1. جداول بدون RLS مفعّل SELECT '⚠️ جداول بدون RLS' as issue, t.tablename FROM pg_tables t LEFT JOIN pg_class c ON t.tablename = c.relname AND c.relnamespace = 'public'::regnamespace WHERE t.schemaname = 'public' AND (c.relrowsecurity IS NULL OR c.relrowsecurity = false); -- 2. جداول بدون سياسات SELECT '⚠️ جداول بدون سياسات' as issue, t.tablename FROM pg_tables t WHERE t.schemaname = 'public' AND NOT EXISTS (SELECT 1 FROM pg_policies p WHERE p.schemaname = 'public' AND p.tablename = t.tablename); -- 3. جداول ب tenant_id بدون فهرس SELECT '⚠️ جداول بـ tenant_id بدون فهرس' as issue, c.table_name FROM information_schema.columns c WHERE c.table_schema = 'public' AND c.column_name = 'tenant_id' AND NOT EXISTS ( SELECT 1 FROM pg_indexes i WHERE i.schemaname = 'public' AND i.tablename = c.table_name AND i.indexdef LIKE '%tenant_id%' );
